🔍 金十页面调试工具

使用说明

  1. 打开金十网站 (https://www.jin10.com/flash)
  2. 按 F12 打开开发者工具
  3. 切换到 "Console" 标签
  4. 复制下面的代码，粘贴到控制台，按回车执行

调试代码（复制到金十页面的控制台）

// ======================================== // 金十页面DOM结构调试脚本 // ======================================== console.log('=== 金十页面调试开始 ===\n'); // 1. 查找所有可能的快讯容器 console.log('📌 步骤1: 查找快讯容器'); const possibleContainers = [ 'jin10-flash-item', 'flash-item', 'flash', 'news', 'item', 'list-item', 'content-item' ]; possibleContainers.forEach(selector => { const elements = document.querySelectorAll(`.${selector}`); if (elements.length > 0) { console.log(`✓ 找到 .${selector}: ${elements.length} 个`); console.log(' 示例:', elements[0]); } }); // 2. 查找所有包含时间的元素 console.log('\n📌 步骤2: 查找时间元素'); const timeElements = document.querySelectorAll('[class*="time"], [class*="date"]'); console.log(`✓ 找到时间相关元素: ${timeElements.length} 个`); if (timeElements.length > 0) { console.log(' 示例:', timeElements[0]); } // 3. 查找所有可能包含快讯内容的元素 console.log('\n📌 步骤3: 查找内容元素'); const contentElements = document.querySelectorAll('[class*="content"], [class*="text"], [class*="title"]'); console.log(`✓ 找到内容相关元素: ${contentElements.length} 个`); // 4. 分析页面结构 console.log('\n📌 步骤4: 分析页面主容器'); const mainContainers = [ 'main', '[class*="container"]', '[class*="wrapper"]', '[class*="flash"]', '[class*="news"]' ]; mainContainers.forEach(selector => { const elements = document.querySelectorAll(selector); if (elements.length > 0) { console.log(`✓ ${selector}: ${elements.length} 个`); } }); // 5. 输出第一个快讯的完整结构 console.log('\n📌 步骤5: 分析第一个快讯结构'); const firstItem = document.querySelector('[class*="item"], [class*="flash"], [class*="news"]'); if (firstItem) { console.log('第一个快讯元素:'); console.log('HTML:', firstItem.outerHTML.substring(0, 500) + '...'); console.log('Class:', firstItem.className); console.log('Text:', firstItem.textContent.substring(0, 100)); } else { console.log('❌ 未找到快讯元素'); } // 6. 查找所有一级子元素 console.log('\n📌 步骤6: 页面body的直接子元素'); console.log('Body子元素数量:', document.body.children.length); Array.from(document.body.children).forEach((child, i) => { console.log(` ${i + 1}. ${child.tagName}.${child.className}`); }); console.log('\n=== 调试完成 ===');
